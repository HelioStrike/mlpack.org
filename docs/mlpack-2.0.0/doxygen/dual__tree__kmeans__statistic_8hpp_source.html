63  for (size_t i = 0; i < node.NumPoints(); ++i)
64  {
65  // Correct handling of cover tree: don't double-count the point which
66  // appears in the children.
67  if (tree::TreeTraits<TreeType>::HasSelfChildren && i == 0 &&
68  node.NumChildren() > 0)
69  continue;
70  centroid += node.Dataset().col(node.Point(i));
71  }
72 
73  for (size_t i = 0; i < node.NumChildren(); ++i)
74  centroid += node.Child(i).NumDescendants() *
75  node.Child(i).Stat().Centroid();
76 
77  centroid /= node.NumDescendants();

80  trueChildren.resize(node.NumChildren());
81  for (size_t i = 0; i < node.NumChildren(); ++i)

83  }
