187 inline void GaussianDistribution::LogProbability(const arma::mat& x,
188  arma::vec& logProbabilities) const
189 {
190  // Column i of 'diffs' is the difference between x.col(i) and the mean.
191  arma::mat diffs = x - (mean * arma::ones<arma::rowvec>(x.n_cols));
192 
193  // Now, we only want to calculate the diagonal elements of (diffs' * cov^-1 *
194  // diffs). We just don't need any of the other elements. We can calculate
195  // the right hand part of the equation (instead of the left side) so that
196  // later we are referencing columns, not rows -- that is faster.
197  const arma::mat rhs = -0.5 * invCov * diffs;
198  arma::vec logExponents(diffs.n_cols); // We will now fill this.
199  for (size_t i = 0; i < diffs.n_cols; i++)
200  logExponents(i) = accu(diffs.unsafe_col(i) % rhs.unsafe_col(i));
201 
202  const size_t k = x.n_rows;
203 
204  logProbabilities = -0.5 * k * log2pi - 0.5 * logDetCov + logExponents;
205 }
