98  inline void WUpdate(const MatType& V,
99  arma::mat& W,
100  const arma::mat& H)
101  {
102  size_t n = V.n_rows;
103  size_t m = V.n_cols;
104 
105  size_t r = W.n_cols;
106 
107  // initialize the momentum of this iteration.
108  mW = momentum * mW;
109 
110  // Compute the step.
111  arma::mat deltaW;
112  deltaW.zeros(n, r);
113  for (size_t i = 0; i < n; i++)
114  {
115  for (size_t j = 0; j < m; j++)
116  {
117  const double val = V(i, j);
118  if (val != 0)
119  deltaW.row(i) += (val - arma::dot(W.row(i), H.col(j))) *
120  arma::trans(H.col(j));
121  }
122  // Add regularization.
123  if (kw != 0)
124  deltaW.row(i) -= kw * W.row(i);
125  }
126 
127  // Add the step to the momentum.
128  mW += u * deltaW;
129  // Add the momentum to the W matrix.
130  W += mW;
131  }
