41 T& GetParam(
42  util::ParamData& d,
43  const typename boost::enable_if<arma::is_arma_type<T>>::type* = 0)
44 {
45  // If the matrix is an input matrix, we have to load the matrix. 'value'
46  // contains the filename. It's possible we could load empty matrices many
47  // times, but I am not bothered by that---it shouldn't be something that
48  // happens.
49  typedef std::tuple<T, typename ParameterType<T>::type> TupleType;
50  TupleType& tuple = *boost::any_cast<TupleType>(&d.value);
51  const std::string& value = std::get<1>(tuple);
52  T& matrix = std::get<0>(tuple);
53  if (d.input && !d.loaded)
54  {
55  // Call correct data::Load() function.
56  if (arma::is_Row<T>::value || arma::is_Col<T>::value)
57  data::Load(value, matrix, true);
58  else
59  data::Load(value, matrix, true, !d.noTranspose);
60  d.loaded = true;
61  }
62 
63  return matrix;
64 }
